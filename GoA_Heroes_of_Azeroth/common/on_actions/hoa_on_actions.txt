# Hooks into game events for loot, battles, captures, and deaths

# Fires for all characters once per year

	}
}

# Fires for all characters once every 5 years
on_five_year_pulse = {
	events = {
	}
}

# Fires when a character wins a major battle
on_battle_won_owner = {
	events = {
		hoa_marshal.11 # Commander Glory - chance for a commander to gain renown
		hoa_loot.3     # Battlefield loot - chance to find items/gold
		hoa_loot.4     # Enemy commander loot - chance to claim their gear
	}
}

# Fires when a character loses a major battle
on_battle_lost_owner = {
	events = {
	}
}

# Fires when a war ends (winner side)
on_war_ended_victory = {
	events = {
		hoa_loot.7 # Demand tribute artifacts from defeated ruler
	}
}

# Fires when a character dies
# ROOT = dying character, FROM = killer (if any)
on_death = {

	}
}
